Determinant#

sifat-sifat Determinant#

  1. Sifat Kali: Determinan dari hasil perkalian dua matriks sama dengan hasil kali determinan masing-masing matriks.

  2. Sifat Transpose: Determinan matriks dan transposenya sama.

  3. Perkalian dengan Skalar: Mengalikan matriks dengan skalar ( k ) mengalikan determinannya dengan ( k^n ).

  4. Determinan Matriks Identitas: Determinan dari matriks identitas selalu 1.

  5. Penambahan Baris atau Kolom: Menambahkan kelipatan skalar dari satu baris (atau kolom) ke baris (atau kolom) lainnya tidak mengubah determinan.

  6. Pertukaran Baris atau Kolom: Menukar dua baris atau kolom pada matriks mengubah tanda determinan.

Kofaktor Matriks#

Menghitung Kofaktor, Determinan dan Invers suatu matriks#

import numpy as np

# Menghitung kofaktor suatu elemen dalam matriks
def cofactor(matrix, i, j):
    return (-1) ** (i + j) * np.linalg.det(np.delete(np.delete(matrix, i, axis=0), j, axis=1))

# Menghitung matriks kofaktor dari matriks input
def cofactor_matrix(matrix):
    return np.array([[cofactor(matrix, i, j) for j in range(matrix.shape[1])] for i in range(matrix.shape[0])])

# Menghitung determinan matriks
def determinant(matrix):
    return np.linalg.det(matrix)

# Contoh matriks
A = np.array([[3,1,3],
              [2,1,2],
              [4,2,1]])

# Menghitung matriks kofaktor
C = cofactor_matrix(A)
print("Matriks Kofaktor:")
print(C)

# Menghitung determinan
det_A = determinant(A)
print("\nDeterminan:", det_A.round())

# Menghitung invers matriks
def inverse(matrix):
    det = determinant(matrix)
    if det == 0:
        raise ValueError("Determinan nol, matriks tidak dapat diinvers")
    else:
        return cofactor_matrix(matrix).T / det

# Menghitung invers dari matriks A
A_inv = inverse(A)
print("\nInvers Matriks A:")
print(A_inv.round(2))

Mengitung determinan dari submatriks A21, A22 dan A23#

import numpy as np

# Matriks 3x3
A = np.array([[3, 1, 3],
              [2, 1, 2],
              [4, 2, 1]])

# Membuat submatriks A21, A22, dan A23
A21 = np.delete(np.delete(A, 0, axis=0), 1, axis=1)
A22 = np.delete(np.delete(A, 0, axis=0), 1, axis=1)
A23 = np.delete(np.delete(A, 0, axis=0), 2, axis=1)

# Menghitung determinan dari setiap submatriks
det_A21 = np.linalg.det(A21)
det_A22 = np.linalg.det(A22)
det_A23 = np.linalg.det(A23)

print("Determinan dari submatriks A21:", det_A21.round())

print("Determinan dari submatriks A22:", det_A22.round())

print("Determinan dari submatriks A23:", det_A23.round())

Determinan dengan Ekspansi Kolom Kofaktor#

import numpy as np

def minor(matriks, i, j):
    """Menghitung minor dari elemen (i, j) dalam matriks."""
    return np.linalg.det(np.delete(np.delete(matriks, i, axis=0), j, axis=1))

def kofaktor(matriks, i, j):
    """Menghitung kofaktor dari elemen (i, j) dalam matriks."""
    return ((-1) ** (i + j)) * minor(matriks, i, j)

def determinan_dengan_kolom(matriks, kolom):
    """Menghitung determinan matriks menggunakan ekspansi kofaktor pada kolom tertentu."""
    n = len(matriks)
    return sum(matriks[i][kolom] * kofaktor(matriks, i, kolom) for i in range(n))

# Contoh penggunaan:
matriks = np.array([[3, 1, 3],
                    [2, 1, 2],
                    [4, 2, 1]])
det_matriks = determinan_dengan_kolom(matriks, 1)  # Menggunakan kolom kedua (indeks 1)
print(f"Determinan matriks dengan menggunakan kolom kedua adalah: {det_matriks.round()}")

Determinan matriks dengan menggunakan kolom kedua adalah: -3.0

import numpy as np

# Membuat matriks 5x5 dengan elemen-elemen yang ditentukan
A = np.array([[5, 0, 3, 3, 5],
              [2, 3, 3, 2, 4],
              [7, 6, 8, 8, 1],
              [6, 7, 7, 4, 1],
              [5, 4, 5, 6, 4]])

print("Matriks A:")
for row in A:
    print(row)

# Menghitung determinan matriks
determinant = np.linalg.det(A)

print("\nDeterminan dari matriks A:", determinant.round())

# Menghitung invers matriks menggunakan numpy
try:
    matrix_inv = np.linalg.inv(A)
    print("\nInvers Matriks A:")
    print(matrix_inv.round(2))
except np.linalg.LinAlgError:
    print("\nDeterminan nol, matriks tidak dapat diinvers")
